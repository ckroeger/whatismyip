    function setupPayments(plan, price) {
      current_plan = plan;
      current_price = price;
      var plans = ["monthly","yearly","sixmonthly"];
      var processors = ['credit_card', 'paypal', 'amazon', 'bitpay', 'cashu', 'okpay', 'shapeshift'];
      $.each(plans, function(_i, value) {
        $('#terms_' + value).hide();
        $('#prepayment_terms_' + value).hide();
        $.each(processors, function(_j, processor) {
          $('#' + processor + '_form_' + value).hide();
        });
        $('.paymentwall-payment-method').hide();
      });
      $.each(processors, function(_j, processor) {
        $('#terms_' + plan).show();
        $('#prepayment_terms_' + plan).show();
        $('#' + processor + '_form_' + plan).show();
        $('.paymentwall-payment-method-' + plan).show();
      });

      switch($('#prepayment_processor').val()) {
        case 'bitpay':
          $('#prepayment_form').attr('action', bitpay_links[plan]);
          break;
        case 'cashu':
          $('#prepayment_form').attr('action', cashu_links[plan]);
          break;
        case 'okpay':
          $('#prepayment_form').attr('action', okpay_links[plan]);
          break;
        case 'shapeshift':
          $('#prepayment_form').attr('action', shapeshift_links[plan]);
          break;
      }
      $("#navigate-away").hide();
    }
